IF(NOT MPEG2_NAMESPACE)
  SET(MPEG2_NAMESPACE "MPEG2")
  SET(MPEG2_STANDALONE 1)
ENDIF(NOT MPEG2_NAMESPACE)
# In all cases:
STRING(TOLOWER ${MPEG2_NAMESPACE} MPEG2_LIBRARY_NAME)

PROJECT(${MPEG2_NAMESPACE} C)
CMAKE_MINIMUM_REQUIRED(VERSION 2.2)
SET(BUILD_SHARED_LIBS ON)

#-----------------------------------------------------------------------------
# MPEG2 version number
SET(MPEG2_MAJOR_VERSION 1)
SET(MPEG2_MINOR_VERSION 2)
SET(MPEG2_BUILD_VERSION 0)
SET(MPEG2_VERSION 
      "${MPEG2_MAJOR_VERSION}.${MPEG2_MINOR_VERSION}.${MPEG2_BUILD_VERSION}")

SUBDIRS(src)
